[docs]    def run(self):
        with open(self.filename + '.py', 'w') as f:
            f.write(self.header)
            f.write(self.matl)
            f.write(self.geo)
            f.write(self.render)
        subprocess.Popen(['blender --python=%s --background' % \
            (self.filename + '.py')], shell=True)
